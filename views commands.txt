---table creation

CREATE TABLE StudentDetails (
    S_ID INT PRIMARY KEY,
    Name VARCHAR(50),
    Address VARCHAR(100)
);

CREATE TABLE StudentMarks (
    ID INT PRIMARY KEY,
    Name VARCHAR(50),
    Marks INT,
    Age INT
);

---create views

CREATE VIEW DetailsView AS
SELECT Name, Address FROM StudentDetails
WHERE S_ID < 5;

--query for view

SELECT * FROM DetailsView;

--View joining StudentDetails and StudentMarks:

CREATE VIEW MarksView AS
SELECT StudentDetails.Name, StudentDetails.Address, StudentMarks.Marks
FROM StudentDetails, StudentMarks
WHERE StudentDetails.Name = StudentMarks.Name;

--Query the joined view
SELECT * FROM MarksView;


--Inserting Records into a View
INSERT INTO StudentDetails (S_ID, Name, Address)
VALUES (6, 'Frank', '303 Cedar Street');


--Updating Records through a View
UPDATE DetailsView
SET Address = '404 Elm Street'
WHERE Name = 'Alice';


--Deleting Records through a View
DELETE FROM DetailsView
WHERE Name = 'Bob';

--Dropping a View
DROP VIEW MarksView;
